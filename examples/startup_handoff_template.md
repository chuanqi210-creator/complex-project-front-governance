```yaml

  objective: ""
  primary_claim: ""
  claim_readiness: idea_or_candidate

current_basis:
  included: []
  excluded: []
  unknown: []

  - claim_ladder
  - evidence_contract
  - execution_validation
  - transfer_boundary

  selected_now: []
  rejected_now: []
  backlog: []
  next_reconsideration_trigger:
    - stage_transition
    - blocked_verification
    - before_final_claim

micro_validation:
  task: ""
  concrete_artifact_or_record: ""
  direct_observed: []
  unknown_or_not_verified: []
  result: ""
  downgrade_rule: ""

risk_boundary:
  high_risk_tags: []
  external_write: none
  manual_action_required: []
  cannot_conclude: []

decisions:
  - decision: ""
    reason: ""
    alternatives_rejected: []

next_route:
  action: ""
  why_next: ""
  stop_condition: ""
```
